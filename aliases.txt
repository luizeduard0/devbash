# devbash
Helpful Dev Bashes

# General
alias htdocs="/path/to/htdocs"
alias mysql="/path/to/mysql"

# Laravel
alias art="php artisan"
alias migrate="php artisan migrate"
alias fresh="php artisan migrate:refresh -seed"
alias serve="php artisan serve serve"
alias migrate="php artisan migrate"
alias rback="php artisan migrate:rollback"
alias model="php artisan make:model -m "
alias migration="php artisan make:migration"

# PHPUnit
alias t="phpunit"

# Composer
alias cr="composer require"
alias cpr="composer.phar require"
alias cdo="composer dumpautoload -o"
alias ci="composer install"
alias cu="composer update"

# Git
alias ga="git add"
alias gc="git commit -m"
alias gp="git pull"
alias gcb="git checkout"
alias gnb="git checkout -b "
alias gri="git rm -r --cached ."
alias ghr="git reset --hard"
